  server_name  large_screen_server;

    root   /app/web/project-config/dist;
    index  index.html index.htm;

  # 大屏编辑器后端代理
  location ^~ /bigScreenApi/ {
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;

    proxy_pass http://10.168.4.14:9999/;
    # 网关
    # proxy_pass http://10.168.4.210:9080/mti-large-screen/;

  }

  # 统一基础平台相关代理
  location ^~ /mti-service-system/ {
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;

    proxy_pass http://10.168.4.14:19040/;
    # 网关
    # proxy_pass http://10.168.4.210:9080/mti-service-platform/;
  }
  location ^~ /mti-service-uum/ {
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;

    # 网关
    # proxy_pass http://10.168.4.210:9080/mti-service-platform/;

  location ^~ /mti-service-sockjs/ {
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_pass http://10.168.4.14:19040/;

  # 表单引擎产品服务代理
