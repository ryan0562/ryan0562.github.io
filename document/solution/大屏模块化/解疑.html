# 模板、模型、工程数据如何交互的

  1. 在编辑器项目public/index.html内添加了全局通用库
<!-- vue -->
  <script src="/static/bigScreen/libs/vue.js"></script>
  <!-- vuex -->
  <script src="/static/bigScreen/libs/vuex.js"></script>
  <!-- vue-router -->
  <script src="/static/bigScreen/libs/vue-router.js"></script>
  <!-- axios -->
  <script src="/static/bigScreen/libs/axios.js"></script>
  <!-- element-ui -->
  <link rel="stylesheet" href="/static/bigScreen/libs/element-ui/index.css">
  <script src="/static/bigScreen/libs/element-ui/index.js"></script>
  <script type="module">
    Vue.prototype.$ELEMENT = {
      size: 'small'
    };
    </script>
    <!-- editor -->
  <script src="/bigScreen/libs/editor/editor.umd.min.js"></script>
  <script type="module">
    Vue.use(editor)
  </script>
  1. 在所有项目的vue.config.js内过滤掉通用库，包括（模板、模型、工程）
configureWebpack: {
  ...,
  externals: {
    vue: 'Vue',
    vuex: 'Vuex',
    'element-ui': 'ELEMENT',
    'vue-router': 'VueRouter',
    'js-cookie': 'Cookies',
    axios: 'axios',
    '@form-create/element-ui': 'formCreate'
  }
}
  1. 基于VUEX来管理全局数据
state: {
  // 是否开发模板
  developmentTemplates: true,
  /* 控制器 */
  config: {},
  /* 大屏模型 */
  bigScreenModel: {
    model: [],
    template: {}
  }
}
