# ElementUI同步var换肤解决方案

github (opens new window) 不同的tag是不同的用法

# 使用scss修改变量

// var.scss
$--font-path: '~element-ui/lib/theme-chalk/fonts';

[data-theme="light"] {
  $--color-primary: #33af92; 
  --color-primary: #{$--color-primary};

  @import "~element-ui/packages/theme-chalk/src/index.scss";
}

[data-theme="dark"] {
  $--color-primary: #9b2200;
  --color-primary: #{$--color-primary};
 
  @import "~element-ui/packages/theme-chalk/src/index.scss";
}


// main.js
import Vue from 'vue';
import Element from 'element-ui';
import './var.less'
Vue.use(Element, { size: 'small', zIndex: 3000 });

var调用scss变量,生成多套CSS实现。

  • 优点：简单易用，后期维护成本低
  • 缺点：多套css占用内存，需要使用到scss

# 动态link换肤

// main.js
import Vue from 'vue';
import Element from 'element-ui';
import './var.less'

let theme = 'light'

var head = document.getElementsByTagName('HEAD').item(0);
var style = document.createElement('link');
style.href = `/theme/${theme}/index.css`;
style.rel = 'stylesheet';
style.type = 'text/css';
head.appendChild(style);

Object.defineProperty(window, '_theme', {
  get() { return theme; },
  set(val) {
    theme = val;
    style.href = `/theme/${theme}/index.css`;
    document.documentElement.setAttribute("data-theme", theme);
  }
});


Vue.use(Element, { size: 'small', zIndex: 3000 });

// var.less
[data-theme="light"] {
  --color-primary: #00439b;
}

[data-theme="dark"] {
  --color-primary: #9b2200;
}

解耦elementui与var，使用官方皮肤构建工具 (opens new window)（或者官方的npm插件），下载配置好的主题样式文件，放入public/theme/${theme}文件夹下，通过监听window._theme动态加载link来实现。

  • 优点：内存低，var跟elementui解耦，自由度高
  • 缺点：上手难度高，后期维护成本高
