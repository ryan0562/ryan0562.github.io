# 权限指令

# 基础用法

 <template>
  <el-button v-permission="['xxx']">aaa</el-button>
 </template>

# 表格用法

 <template>
  <el-table>
    // 表格上添加参数elTableColumn
    <el-table-column  v-permission:elTableColumn="['xxx']"></el-table-column>
  </el-table>
 </template>

# 指令代码

import store from '@/store'
import Vue from 'vue'

// 校验用户权限，传入一个数组
function hasPermission(role) {
  return role.some(item => {
    return store.state.permission.roles.includes(item)
  })
}

// 注册一个全局自定义指令 `v-permission` 判断该角色是否有权限
Vue.directive('permission', {
  // 被绑定元素插入父节点时
  inserted: (el, { value, arg }, vnode) => {
    // 如果没有权限就移除此节点
    if (!hasPermission(value)) {
      // // 存在参数时
      if (arg) {
        // 指令适配el-table-column
        if (arg === 'elTableColumn') {
          el.__vue__.$destroy()
          return
        }
      }
      el.parentNode.removeChild(el);
    }
  }
})
